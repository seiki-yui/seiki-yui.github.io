@Handler
suspend fun UserCommandSender.handle(number: Int = 1, line: String = "r18=0") {
	if (number in 1..10) {
		SeikiMain.logger.info { "Setu - Start caching..." }
		repeat(number) { now ->
			kotlin.runCatching {
				val url = "https://api.lolicon.app/setu/v2?size=regular&$line"
				val result = SweetBoy.get(url).use { it.body()!!.string() }
				val json = Gson().fromJson(result, LoliconApi::class.java)
				val regular = json.data.first().urls.regular
				val message = subject.uploadAsImage(regular).sendTo(subject)
				launch {
					delay((20..40).random() * 1000L)
					SeikiMain.logger.info { "Setu - ${now + 1} / $number RECALL" }
					message.recall()
				}
			}.onSuccess {
				SeikiMain.logger.info { "Setu - ${now + 1} / $number SUCCESS" }
			}.onFailure {
				SeikiMain.logger.warning { "Setu - ${now + 1} / $number FAILURE" }
			}
		}
	}
}

fsys.enum("~\Players", {"*.png", "*.jpg"},
	function(dirname, filename, fullpath, _) { 
		if filename { 
			var regex = preg("(?P<fname>[\s\S]+)(?P<ext>\.[\s\S]+)")
			var m = regex.exec(filename)
			var bitMap = gdip.bitmap(dirname + m.fname + m.ext) 
			var bmpdata = bitMap.lockData32()
			for(h = 1; bitMap.height){
				for(w = 1; bitMap.width){  
					if bmpdata.bits.rows[h].pixels[w] > 16777215 
					select bmpdata.bits.rows[h].pixels[w] % 16777216 {
						case == 0 
							bmpdata.bits.rows[h].pixels[w] = 0xFF010101
						case == 16777215 
							bmpdata.bits.rows[h].pixels[w] = 0xFFFEFEFE
					}
				} 
			}
			bitMap.unlockData(bmpdata)
			var timestamp = time().millisecond
			bitMap.save(dirname ++ timestamp ++ m.ext)
			bitMap.dispose()
			fsys.delete(dirname ++ m.fname ++ m.ext)
			fsys.rename(dirname ++ timestamp ++ m.ext,dirname ++ m.fname ++ m.ext)
			console.log("处理中", fullpath)
		}
	}
)
